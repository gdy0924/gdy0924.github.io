12    #四次（卷积+上采样）：一次上采样2倍
13    def _init_decode(self):
14        extra_in_channels = int(self.embedding_dim / 4)
15        in_channels = [
16            self.embedding_dim,
17            extra_in_channels,
18            extra_in_channels,
19            extra_in_channels,
20            extra_in_channels,
21        ]
22        out_channels = [
23            extra_in_channels,
24            extra_in_channels,
25            extra_in_channels,
26            extra_in_channels,
27            self.num_classes,
28        ]
29
30        modules = []
31        for i, (in_channel, out_channel) in enumerate(zip(in_channels, out_channels)):
32            modules.append(nn.Conv2d(
33                in_channels=in_channel,
34                out_channels=out_channel,
35                kernel_size=1,
36                stride=1,
37                padding=self._get_padding('VALID', (1, 1),),
38                )
39            )
40            #进行4次上采样
41            if i != 4:
42                modules.append(nn.Upsample(scale_factor=2, mode='bilinear'))
43        self.decode_net = IntermediateSequential(*modules, return_intermediate=False)
44
45    def decode(self, x, intmd_x, intmd_layers=None):
46        x = self._reshape_output(x)
47        x = self.decode_net(x)
48        return x

 1class SETR_MLA(nn.Module):
 2    def __init__(self, in_channels=1024, mla_channels=256, norm_cfg=None):
 3        super(Conv_MLA, self).__init__()
 4        #每层特征经过1×1卷积
 5        self.mla_p2_1x1 = nn.Sequential(nn.Conv2d(
 6            in_channels, mla_channels, 1, bias=False), build_norm_layer(norm_cfg, mla_channels)[1], nn.ReLU())
 7        self.mla_p3_1x1 = nn.Sequential(nn.Conv2d(
 8            in_channels, mla_channels, 1, bias=False), build_norm_layer(norm_cfg, mla_channels)[1], nn.ReLU())
 9        self.mla_p4_1x1 = nn.Sequential(nn.Conv2d(
10            in_channels, mla_channels, 1, bias=False), build_norm_layer(norm_cfg, mla_channels)[1], nn.ReLU())
11        self.mla_p5_1x1 = nn.Sequential(nn.Conv2d(
12            in_channels, mla_channels, 1, bias=False), build_norm_layer(norm_cfg, mla_channels)[1], nn.ReLU())
13        self.mla_p2 = nn.Sequential(nn.Conv2d(mla_channels, mla_channels, 3, padding=1,
14                                    bias=False), build_norm_layer(norm_cfg, mla_channels)[1], nn.ReLU())
15        self.mla_p3 = nn.Sequential(nn.Conv2d(mla_channels, mla_channels, 3, padding=1,
16                                    bias=False), build_norm_layer(norm_cfg, mla_channels)[1], nn.ReLU())
17        self.mla_p4 = nn.Sequential(nn.Conv2d(mla_channels, mla_channels, 3, padding=1,
18                                    bias=False), build_norm_layer(norm_cfg, mla_channels)[1], nn.ReLU())
19        self.mla_p5 = nn.Sequential(nn.Conv2d(mla_channels, mla_channels, 3, padding=1,
20                                    bias=False), build_norm_layer(norm_cfg, mla_channels)[1], nn.ReLU())
21
22    def to_2D(self, x):
23        n, hw, c = x.shape
24        h = w = int(math.sqrt(hw))
25        x = x.transpose(1, 2).reshape(n, c, h, w)
26        return x

28    def forward(self, res2, res3, res4, res5):
29        #reshape
30        res2 = self.to_2D(res2)
31        res3 = self.to_2D(res3)
32        res4 = self.to_2D(res4)
33        res5 = self.to_2D(res5)
34
35        #1×1卷积
36        mla_p5_1x1 = self.mla_p5_1x1(res5)
37        mla_p4_1x1 = self.mla_p4_1x1(res4)
38        mla_p3_1x1 = self.mla_p3_1x1(res3)
39        mla_p2_1x1 = self.mla_p2_1x1(res2)
40
41        #下层特征图与上层特征图相加
42        mla_p4_plus = mla_p5_1x1 + mla_p4_1x1
43        mla_p3_plus = mla_p4_plus + mla_p3_1x1
44        mla_p2_plus = mla_p3_plus + mla_p2_1x1
45
46        #3×3卷积
47        mla_p5 = self.mla_p5(mla_p5_1x1)
48        mla_p4 = self.mla_p4(mla_p4_plus)
49        mla_p3 = self.mla_p3(mla_p3_plus)
50        mla_p2 = self.mla_p2(mla_p2_plus)
51
52        return mla_p2, mla_p3, mla_p4, mla_p5
