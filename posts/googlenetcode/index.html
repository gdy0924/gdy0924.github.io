@Gao dy

    GoogLeNet获得了2014年ILSVRC比赛分类任务的冠军，其利用Inception模块，在加深网络深度的同时，减少了参数量，从而减少计算资源的利用。

实现

基于pytorch实现的代码如下：

 1import torch
 2import torch.nn as nn
 3import torch.nn.functional as F
 4
 5#卷积+BN+ReLu
 6class BasicConv2d(nn.Module):
 7    def __init__(self, in_channels, out_channals, **kwargs):
 8        super(BasicConv2d, self).__init__()
 9        self.conv = nn.Conv2d(in_channels, out_channals, **kwargs)
10        self.bn = nn.BatchNorm2d(out_channals)
11
12    def forward(self, x):
13        x = self.conv(x)
14        x = self.bn(x)
15        return F.relu(x)
16
17# Inception模块
18class Inception(nn.Module):
19    #（输入通道数，1×1卷积的输出通道数，3×3卷积的降维通道数，3×3卷积的输出通道数，
20    #5×5卷积的降维通道数，5×5卷积的输出通道数，池化通道数）
21    def __init__(self, in_planes,n1x1, n3x3red, n3x3, n5x5red, n5x5, pool_planes):
22        super(Inception, self).__init__()
23        # 1x1 conv branch
24        self.b1 = BasicConv2d(in_planes, n1x1, kernel_size=1)
25
26        # 1x1 conv -> 3x3 conv branch
27        self.b2_1x1_a = BasicConv2d(in_planes, n3x3red, kernel_size=1)
28        self.b2_3x3_b = BasicConv2d(n3x3red, n3x3, kernel_size=3, padding=1)
29
30        # 1x1 conv -> 3x3 conv -> 3x3 conv branch
31        #利用两个3x3实现5×5
32        self.b3_1x1_a = BasicConv2d(in_planes, n5x5red, kernel_size=1)
33        self.b3_3x3_b = BasicConv2d(n5x5red, n5x5, kernel_size=3, padding=1)
34        self.b3_3x3_c = BasicConv2d(n5x5, n5x5, kernel_size=3, padding=1)
35
36        # 3x3 pool -> 1x1 conv branch
37        self.b4_pool = nn.MaxPool2d(3, stride=1, padding=1)
38        self.b4_1x1 = BasicConv2d(in_planes, pool_planes, kernel_size=1)
39
40    def forward(self, x):
41        y1 = self.b1(x)
42        y2 = self.b2_3x3_b(self.b2_1x1_a(x))
43        y3 = self.b3_3x3_c(self.b3_3x3_b(self.b3_1x1_a(x)))
44        y4 = self.b4_1x1(self.b4_pool(x))
45        # y的维度为[batch_size, out_channels, C_out,L_out]
46        # 合并不同卷积下的特征图
47        return torch.cat([y1, y2, y3, y4], 1)
48
49class GoogLeNet(nn.Module):
50    def __init__(self):
51        super(GoogLeNet, self).__init__()
52        self.pre_layers = BasicConv2d(3, 192, kernel_size=3, padding=1)
53
54        self.a3 = Inception(192,  64,  96, 128, 16, 32, 32)
55        self.b3 = Inception(256, 128, 128, 192, 32, 96, 64)
56
57        self.maxpool = nn.MaxPool2d(3, stride=2, padding=1)
58
59        self.a4 = Inception(480, 192,  96, 208, 16,  48,  64)
60        self.b4 = Inception(512, 160, 112, 224, 24,  64,  64)
61        self.c4 = Inception(512, 128, 128, 256, 24,  64,  64)
62        self.d4 = Inception(512, 112, 144, 288, 32,  64,  64)
63        self.e4 = Inception(528, 256, 160, 320, 32, 128, 128)
64
65        self.a5 = Inception(832, 256, 160, 320, 32, 128, 128)
66        self.b5 = Inception(832, 384, 192, 384, 48, 128, 128)
67
68        self.avgpool = nn.AvgPool2d(8, stride=1)
69        self.linear = nn.Linear(1024, 10)
70
71    def forward(self, x):
72        out = self.pre_layers(x)
73        out = self.a3(out)
74        out = self.b3(out)
75        out = self.maxpool(out)
76        out = self.a4(out)
77        out = self.b4(out)
78        out = self.c4(out)
79        out = self.d4(out)
80        out = self.e4(out)
81        out = self.maxpool(out)
82        out = self.a5(out)
83        out = self.b5(out)
84        out = self.avgpool(out)
85        out = out.view(out.size(0), -1)
86        out = self.linear(out)
87        return out
