 1import torch 
 2import torch.nn as nn
 3import torch.nn.functional as F
 4
 5# 使用torch的nn模块中BatchNorm/在encoding文件中定义的BatchNorm
 6try:
 7    from encoding.nn import SyncBatchNorm
 8    _BATCH_NORM = SyncBatchNorm
 9except:
10    _BATCH_NORM = nn.BatchNorm2d
11_BOTTLENECK_EXPANSION = 4
12
13#卷积+BN+非线性激活
14class _ConvBnReLU(nn.Sequential):
15    BATCH_NORM = _BATCH_NORM
16    def __init__(self, in_ch, out_ch, kernel_size, stride, padding, dilation, relu=True):
17        super(_ConvBnReLU, self).__init__()
18        self.add_module(
19            "conv",nn.Conv2d(in_ch, out_ch, kernel_size, stride, padding, dilation, bias=False),
20        )
21        self.add_module("bn", _BATCH_NORM(out_ch, eps=1e-5, momentum=0.999))
22
23        if relu:
24            self.add_module("relu", nn.ReLU())
25
26#ResNet的Bottleneck
27#1×1卷积降维+3×3卷积+1×1卷积升维（shortcut）
28class _Bottleneck(nn.Module):
29    def __init__(self, in_ch, out_ch, stride, dilation, downsample):
30        super(_Bottleneck, self).__init__()
31        mid_ch = out_ch // _BOTTLENECK_EXPANSION
32        self.reduce = _ConvBnReLU(in_ch, mid_ch, 1, stride, 0, 1, True)
33        self.conv3x3 = _ConvBnReLU(mid_ch, mid_ch, 3, 1, dilation, dilation, True)
34        self.increase = _ConvBnReLU(mid_ch, out_ch, 1, 1, 0, 1, False)
35        self.shortcut = (
36            _ConvBnReLU(in_ch, out_ch, 1, stride, 0, 1, False)
37            if downsample
38            else lambda x: x  # identity
39        )
40    def forward(self, x):
41        h = self.reduce(x)
42        h = self.conv3x3(h)
43        h = self.increase(h)
44        h += self.shortcut(x)
45        return F.relu(h)
46
47#DeepLab由ResLayer堆叠形成
48class _ResLayer(nn.Sequential):
49    def __init__(self, n_layers, in_ch, out_ch, stride, dilation, multi_grids=None):
50        super(_ResLayer, self).__init__()
51        if multi_grids is None:
52            multi_grids = [1 for _ in range(n_layers)]
53        else:
54            assert n_layers == len(multi_grids)
55
56        #下采样在每个ResLayer的第一个block中
57        for i in range(n_layers):
58            self.add_module(
59                "block{}".format(i + 1),
60                _Bottleneck(
61                    in_ch=(in_ch if i == 0 else out_ch),
62                    out_ch=out_ch,
63                    stride=(stride if i == 0 else 1),
64                    dilation=dilation * multi_grids[i],
65                    downsample=(True if i == 0 else False),
66                ),
67            )
68
69#先用7*7的卷积，增大感受野，padding方式设为same，大小不变
70class _Stem(nn.Sequential):
71    def __init__(self, out_ch):
72        super(_Stem, self).__init__()
73        self.add_module("conv1", _ConvBnReLU(3, out_ch, 7, 2, 3, 1))
74        self.add_module("pool", nn.MaxPool2d(3, 2, 1, ceil_mode=True))
75
76# 定义DeepLabV1的网络结构
77class DeepLabV1(nn.Sequential):
78    def __init__(self, n_classes, n_blocks):
79        super(DeepLabV1, self).__init__()
80        ch = [64 * 2 ** p for p in range(6)]
81        self.add_module("layer1", _Stem(ch[0]))
82        self.add_module("layer2", _ResLayer(n_blocks[0], ch[0], ch[2], 1, 1))
83        self.add_module("layer3", _ResLayer(n_blocks[1], ch[2], ch[3], 2, 1))
84        self.add_module("layer4", _ResLayer(n_blocks[2], ch[3], ch[4], 1, 2))
85        self.add_module("layer5", _ResLayer(n_blocks[3], ch[4], ch[5], 1, 4))
86        self.add_module("fc", nn.Conv2d(2048, n_classes, 1))
87
88#构建DeepLab V1模型
89if __name__ == "__main__":
90    model = DeepLabV1(n_classes=21, n_blocks=[3, 4, 23, 3])
91    image = torch.randn(1, 3, 513, 513)
92    print(model)
93    print("input:", image.shape)
94    print("output:", model(image).shape)

 1#ASPP模块
 2class _ASPP(nn.Module):
 3    def __init__(self, in_ch, out_ch, rates):
 4        super(_ASPP, self).__init__()
 5        for i, rate in enumerate(rates):
 6            self.add_module(
 7                "c{}".format(i),
 8                nn.Conv2d(in_ch, out_ch, 3, 1, padding=rate, dilation=rate, bias=True),
 9            )
10
11        for m in self.children():
12            nn.init.normal_(m.weight, mean=0, std=0.01)
13            nn.init.constant_(m.bias, 0)
14
15    def forward(self, x):
16        return sum([stage(x) for stage in self.children()])

 1# 全局平均池化+256个通道的1*1卷积中+双线性上采样
 2class _ImagePool(nn.Module):
 3    def __init__(self, in_ch, out_ch):
 4        super().__init__()
 5        self.pool = nn.AdaptiveAvgPool2d(1)
 6        self.conv = _ConvBnReLU(in_ch, out_ch, 1, 1, 0, 1)
 7
 8    def forward(self, x):
 9        _, _, H, W = x.shape
10        h = self.pool(x)
11        h = self.conv(h)
12        h = F.interpolate(h, size=(H, W), mode="bilinear", align_corners=False)
13        return h
14
15#ASPP模块：新增了1*1卷积+全局池化。
16class _ASPP(nn.Module):
17    def __init__(self, in_ch, out_ch, rates):
18        super(_ASPP, self).__init__()
19        self.stages = nn.Module()
20        self.stages.add_module("c0", _ConvBnReLU(in_ch, out_ch, 1, 1, 0, 1))
21        for i, rate in enumerate(rates):
22            self.stages.add_module(
23                "c{}".format(i + 1),
24                _ConvBnReLU(in_ch, out_ch, 3, 1, padding=rate, dilation=rate),
25            )
26        self.stages.add_module("imagepool", _ImagePool(in_ch, out_ch))
27
28    def forward(self, x):
29        return torch.cat([stage(x) for stage in self.stages.children()], dim=1)
