LeNet

26 Dec 2021
@Gao dy

LeNet是很简单的一个经典卷积神经网络，主要用于手写数字识别，所以是一个多分类任务，并且是十个类别。

实现

基于pytorch实现的代码如下：

 1
 2import torch
 3from torch import nn
 4from d2l import torch as d2l
 5
 6class Reshape(torch.nn.Module):
 7    def forward(self,x):
 8        return x.view(-1,1,28,28)
 9
10net = torch.nn.Sequential(
11    Reshape(),nn.Conv2d(1,6,kernel_size=5,padding=2),nn.Sigmoid(),
12    nn.AvgPool2d(kernel_size=2,stride=2),
13    nn.Conv2d(6,16,kernel_size=5),nn.Sigmoid(),
14    nn.AvgPool2d(kernel_size=2,stride=2),nn.Flatten(),
15    nn.Linear(16*5*5,120),nn.Sigmoid(),
16    nn.Linear(120,84),nn.Sigmoid(),
17    nn.Linear(84,10))
18
19X = torch.rand(size=(1,1,28,28),dtype=torch.float32)
20for layer in net:
21    X = layer(X)
22    print(layer.__class__.__name__,'output shape：\t',X.shape)
23
24batch_size=256
25train_iter,test_iter = d2l.load_data_fashion_mnist(batch_size=batch_size)
26
27def evaluate_accuracy_gpu(net,data_iter,device=None):
28    if isinstance(net, nn.Module):
29        net.eval()
30        if not device:
31            device = next(iter(net.parameters())).device
32    metric = d2l.Accumulator(2)
33    with torch.no_grad():
34        for X,y in data_iter:
35            if isinstance(X,list):
36                X=(x.to(device) for x in X)
37            else:
38                X=X.to(device)
39            y=y.to(device)
40            metric.add(d2l.accuracy(net(X), y), y.numel())
41    return metric[0] / metric[1]
42
43def train(net,train_iter,test_iter,num_epochs,lr,device):
44    def init_weights(m):
45        if type(m) == nn.Linear or type(m) == nn.Conv2d:
46            nn.init.xavier_uniform_(m.weight)
47    net.apply(init_weights)
48    print('training on',device)
49    net.to(device)
50    optimizer=torch.optim.SGD(net.parameters(),lr=lr)
51    loss=nn.CrossEntropyLoss()
52    animator = d2l.Animator(xlabel='epoch',xlim=[1,num_epochs],legend=['train loss','train acc','test acc'])
53    timer,num_batches=d2l.Timer(),len(train_iter)
54    for epoch in range(num_epochs):
55        metric = d2l.Accumulator(3)
56        net.train()
57        for i,(X,y) in enumerate(train_iter):
58            timer.start()
59            optimizer.zero_grad()
60            X,y = X.to(device),y.to(device)
61            y_hat=net(X)
62            l = loss(y_hat,y)
63            l.backward()
64            optimizer.step()
65            with torch.no_grad():
66                metric.add(l * X.shape[0], d2l.accuracy(y_hat, y), X.shape[0])
67            timer.stop
68            train_l = metric[0]/metric[2]
69            train_acc=metric[1]/metric[2]
70            if (i+1)%(num_batches//5)==0 or i==num_batches- 1:
71                animator.add(epoch+(i+1)/num_batches,(train_l,train_acc,None))
72        test_acc=evaluate_accuracy_gpu(net,test_iter)
73        animator.add(epoch+1,(None,None,test_acc))
74    print(f'loss{train_l:.3f},train acc {train_acc:.3f},'
75          f'test acc{test_acc:.3f}')
76    print(f'{metric[2] * num_epochs / timer.sum():.1f} examples/sec',
77         f'on {str(device)}')
78
79lr,num_epochs=0.9,10
80train(net,train_iter,test_iter,num_epochs,lr,d2l.try_gpu())
